### 난생처음 만드는 Vite 플러그인

- vite 플러그인의 구조
```javascript
{
  name: "...",
  resolveId(source, importer, options) {
    // 플러그인의 타겟이 맞는지 확인
    return "..."
  },
  load(id) {
    // 해당 파일을 로드
    return "..."
  }
  transform(code, id) {
    // 해방 파일을 변환
    return "..."
  ]
}
```

### Virtual Module
```javascript
// vite.config.js

function buildTimeLibrary(factories) {
  return {
    name: "virtual-module",
    resolveId(source, importer, options) {
      if (source === "virtual-module") {
        return source;
      }
    },
    load(id) {
      if (id === "virtual-module") {
        return factories.map((factory) => factory()).join("\n");
      }
    }
  }
}

export default {
  plugins: [
    buildTimeLibrary([
      () => `export const name = "Vite"`,
      () => `export const version = "2.0.0"`
    ])
  ]
}
```
